  8 /**
  9 * @class Weather Bridge for fetching JS weather objects.
 10 * @property {String} url The url of the weather data server.
 11 */
 12 function WeatherJSONBridge(url) {
 13     
 14     this.tempurl = url;
 15     
 16     /**
 17     * Fetch a single type of weather data.
 18     * @param {String} resource The type of data to be fetched: 'temperature','humidity','dew point','pressure','wind speed','wind direction','sun hours','rain','max wind speed' are avalible.
 19     * @param {String} datestr The date on which the readings were taken, in the form "YYYY-MM-DD"
 20     * @returns {Object} A JS object representing the requested data.
 21     */
 22     this.getData = function (datestr, resource){
 23         
 24         if(typeof datestr != "string" || typeof resource != "string"){ return -1; }
 25         
 26         switch(resource.toLowerCase()){
 27             case 'temperature':
 28                 var col = 1;
 29                 break;
 30             case 'humidity':
 31                 var col = 2;
 32                 break;
 33             case 'dew point':
 34                 var col = 3;
 35                 break;
 36             case 'pressure':
 37                 var col = 4;
 38                 break;
 39             case 'wind speed':
 40                 var col = 5;
 41                 break;
 42             case 'wind direction':
 43                 var col = 6;
 44                 break;
 45             case 'sun hours':
 46                 var col = 7;
 47                 break;
 48             case 'rain':
 49                 var col = 8;
 50                 break;
 51             case 'max wind speed':
 52                 var col = 9;
 53                 break;
 54             default:
 55                 return -1;
 56                 break;
 57         }
 58         
 59         resource = resource.charAt(0).toUpperCase() + resource.slice(1);
 60         
 61         var readobj = {
 62             "resource":resource,
 63             "units":null,
 64             "description":resource+" at the Computer Lab, Cambridge UK.",
 65             "data":[]
 66             }
 67         
 68         theurl = this.tempurl+datestr;
 69         var readings = $.ajax({ type: "GET", url: theurl,async: false }).responseText.split('\n');
 70         
 71         readobj.units = readings[7].split('\t')[col].replace(/(^\s*)|(\s*$)/gi,"");
 72         
 73         for (i=8; i<readings.length-1;i++){
 74             var reading = readings[i].split('\t')[col];
 75             var time = readings[i].split('\t')[0].split(':');
 76             
 77             var da = datestr.split('-');
 78             var year = parseInt(da[0], 10);
 79             var month = parseInt(da[1], 10)-1;
 80             var day = parseInt(da[2], 10);
 81             var hour = parseInt(time[0], 10);
 82             var min = parseInt(time[1], 10);
 83             
 84             var dateobj = new Date(year,month,day,hour,min);
 85             readobj.data.push([dateobj,reading]);
 86         }
 87         return readobj;
 88     };
 89     
 90     /**
 91     * Fetch all avalible weather data.
 92     * @param {Integer} resource The type of data to be fetched.
 93     * @param {String} datestr The date on which the readings were taken, in the form "YYYY-MM-DD"
 94     * @returns {Object} A JS object representing the requested data.
 95     */
 96     
 97     this.getWeather = function (datestr){
 98         if(typeof datestr != "string"){ return -1; }
 99         
100         var readobj = {
101             "resource":null,
102             "units":null,
103             "description":"Full weather report at the Computer Lab, Cambridge UK.",
104             "data":[]
105             }
106         
107         theurl = this.tempurl+datestr;
108         var readings = $.ajax({ type: "GET", url: theurl,async: false }).responseText.split('\n');
109         
110         readobj.units = readings[7].split('\t');
111         readobj.resource = readings[6].split('\t');
112         
113         for (i=8; i<readings.length-1;i++){
114             var reading = readings[i].split('\t');
115             var time = readings[i].split('\t')[0].split(':');
116             
117             var da = datestr.split('-');
118             var year = parseInt(da[0], 10);
119             var month = parseInt(da[1], 10)-1;
120             var day = parseInt(da[2], 10);
121             var hour = parseInt(time[0], 10);
122             var min = parseInt(time[1], 10);
123             
124             var dateobj = new Date(year,month,day,hour,min);
125             readobj.data.push([dateobj,reading]);
126         }
127         return readobj;
128     };
129 
130 }
